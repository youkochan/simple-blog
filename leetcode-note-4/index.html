199 Binary Tree Right Side View

Given a binary tree, imagine yourself standing on the right side of it, return the values of the nodes you can see ordered from top to bottom.

题目大意是，给你一个二叉树，你站在这棵树的右边，把看到的序列从顶到底输出。一开始我想错了，我以为简单地自顶向下，若有右孩子，则向右移动指针；若没有则左移。后来发现这个简单的想法是错误的。因为有可能左子树的高度大于右子树，因此，向左看的时候看到的最下面一部分是左子树。

由于需要输出一个数组，则可以使用一个buffer，buffer[i]表示i层能看到的数字。

因此，考虑在中序（中右左）递归中记录一个遍历到的最大深度（初始为0），若当前遍历的节点深度大于或者等于最大深度，则将该节点加入buffer（因为在这种情况下该节点一定是最右边的节点，因为遍历顺序的原因，先中，再右，再左）；若深度小于等于最大深度，说明右边有更高的子树（肯定会挡住这个节点，因此这个节点不可见），则跳过这个点，继续搜索这个节点的子节点。

代码如下：

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Return an array of size *returnSize.
 * Note: The returned array must be malloced, assume caller calls free().
 */

void searchTree(struct TreeNode* node, int buffer[], int level, int* maxLevel) {
    if(node == NULL)
        return;
        
    if(level >= (*maxLevel)) {
        buffer[level] = node->val;
        (*maxLevel) += 1;
    }
    
    searchTree(node->right, buffer, level+1, maxLevel);
    searchTree(node->left, buffer, level+1, maxLevel);
}

int* rightSideView(struct TreeNode* root, int* returnSize) {
    int buffer[2000];
    int*result = NULL;
    int count;
    int*maxLevel = (int*)malloc(sizeof(int));
    
    (*maxLevel) = 0;
    
    searchTree(root, buffer, 0, maxLevel);
    
    count = (*maxLevel);
    
    if(count == 0) {
        (*returnSize) = 0;
        
        return NULL;
    }
    else {
        result = (int*)malloc(sizeof(int)*count);
        (*returnSize) = count;
        
        for(count = count-1; count >= 0; count--) {
            result[count] = buffer[count];
        }
        return result;
    }
}

138 Copy List with Random Pointer

A linked list is given such that each node contains an additional random pointer which could point to any node in the list or null. Return a deep copy of the list.

解题思路，使用一个哈希表将待拷贝的链表的节点的地址映射到新地址，同时根据哈希表中有无新地址进行操作，若无新地址，则申请且记录；若有，则使用新地址。代码如下：

/**
 * Definition for singly-linked list with a random pointer.
 * struct RandomListNode {
 *     int label;
 *     RandomListNode *next, *random;
 *     RandomListNode(int x) : label(x), next(NULL), random(NULL) {}
 * };
 */
class Solution {
public:
    RandomListNode *copyRandomList(RandomListNode *head) {
        unordered_map<RandomListNode *, RandomListNode *> hashMap;
        RandomListNode *newHead = nullptr;
        RandomListNode *newTail = nullptr;
        
        while(head) {
            // 地址不在哈希表中
            RandomListNode *tmpNode = nullptr;
            if(hashMap.find(head) == hashMap.end()) {
                tmpNode = new RandomListNode(head->label);
                hashMap[head] = tmpNode;
            }
            // 地址在哈希表中
            else {
                tmpNode = hashMap[head];
                tmpNode->label = head->label;
            }
            
            if(head->random) {
                if(hashMap.find(head->random) == hashMap.end()) {
                    tmpNode->random = new RandomListNode(0);
                    hashMap[head->random] = tmpNode->random;
                }
                else {
                    tmpNode->random = hashMap[head->random];
                }
            }
            
            if(newHead == nullptr) {
                newHead = tmpNode;
                newTail = tmpNode;
            }
            else {
                newTail->next = tmpNode;
                newTail = tmpNode;
            }
            
            head = head->next;
        }
        
        return newHead;
    }
};

68 Text Justification

Given an array of words and a length L, format the text such that each line has exactly L characters and is fully (left and right) justified.

You should pack your words in a greedy approach; that is, pack as many words as you can in each line. Pad extra spaces ‘ ‘ when necessary so that each line has exactly L characters.

Extra spaces between words should be distributed as evenly as possible. If the number of spaces on a line do not divide evenly between words, the empty slots on the left will be assigned more spaces than the slots on the right.

For the last line of text, it should be left justified and no extra space is inserted between words.

For example, words: [“This”, “is”, “an”, “example”, “of”, “text”, “justification.”], L: 16.

Return the formatted lines as:

[
   "This    is    an",
   "example  of text",
   "justification.  "
]

题目要求即为两端对齐单词，思路比较简单，判断单词是否还能加入该字符串，若能，继续加入；若不能，格式化之后加入vector，然后从当前位置开始继续将单词加入字符串。在加入过程中记录字符串中的单词数目以及字符数目，用于计算间隔宽度。最后一行左对齐后直接在尾部补充空格即可。代码如下：

class Solution {
public:
    vector<string> fullJustify(vector<string>& words, int maxWidth) {
        vector<string> result;
        int start = 0;
        int charCount = 0;
        int wordCount = 0;
        int rest = maxWidth;
        // 遍历每一个单词
        for(int i = 0; i < words.size(); i ++) {
            string word = words[i];
            int size = word.size();
            if(rest >= size) {
                rest = rest - size - 1;
                wordCount += 1;
                charCount += size;
            }
            else {
                // 将单词从start填充到i-1
                string buff;
                
                int spaceNum = maxWidth - charCount;
                if(wordCount == 1) {
                    buff = words[start];
                    while(buff.size() < maxWidth) {
                        buff = buff + ' ';
                    }
                }
                else {
                    int div = spaceNum / (wordCount - 1);
                    int mod = spaceNum % (wordCount - 1);
                    buff = buff + words[start];
                    
                    for(int idx = start + 1; idx < i; idx ++) {
                        string gap;
                        int gapLen = (mod == 0) ? div : div + 1;
                        if(mod)
                            mod --;
                        while(gap.size() < gapLen) {
                            gap = gap + ' ';
                        }
                        buff = buff + gap;
                        buff = buff + words[idx];
                    }
                }
                
                result.push_back(buff);
                start = i;
                wordCount = 1;
                charCount = size;
                rest = maxWidth - size - 1;
            }
        }
        
        // 最后一行
        string buff;
        for(int i = start; i < words.size(); i ++) {
            if(i == start)
                buff = buff + words[i];
            else
                buff = buff + ' ' + words[i];
        }
        
        while(buff.size() < maxWidth) {
            buff = buff + ' ';
        }
            
        result.push_back(buff);
        
        return result;
    }
};

75 Sort Colors

Given an array with n objects colored red, white or blue, sort them so that objects of the same color are adjacent, with the colors in the order red, white and blue.

Here, we will use the integers 0, 1, and 2 to represent the color red, white, and blue respectively.

起初看到这个题的时候直接采用了一次遍历计算0，1，2的个数，然后直接一次遍历给数组赋值的做法（简单粗暴），结果竟然都通过了。不得不说Leetcode复杂度检测确实不严格。

第二种做法是采用两个指针，一个指向0的最后一个元素的前一个位置，一个指向2的最后一个元素的后一个位置，顺序扫描数组，遇到0时，将0置换到合适的位置，当前位置加一；遇到1直接跳过；遇到2将2置换到合适的位置，此时当前位置不自增，因为换过来的数字可能为0（如果自增则跳过了该0）。代码如下：

class Solution {
private:
    void swap(vector<int> &nums, int i, int j) {
        int buff = nums[i];
        nums[i] = nums[j];
        nums[j] = buff;
    }

public:
    void sortColors(vector<int>& nums) {
        int l = -1;
        int r = nums.size();
        int p = 0;
        
        while(p < r) {
            if(nums[p] == 0)
                swap(nums, ++l, p++);
            else if(nums[p] == 1)
                p++;
            else
                swap(nums, --r, p);
        }
        
        return;
    }
};

29 Divide Two Integers

Divide two integers without using multiplication, division and mod operator.

If it is overflow, return MAX_INT.

思路比较清晰，即先将除数左移，得到最大的不超过被除数的除数，并记录倍数，当被除数大于除数时，每次从除数中每次减去被除数，并将倍数加入结果，每进行一次将除数与倍数右移一位。

完成代码后发现各种问题，结果发现是最开始我使用while(time>=1)来判断循环是否结束，然而time类型没有设置为unsigned，导致没有进入循环于是出错。另外，除数与被除数使用long long是为了在预判左移后ldivisor不至于溢出。代码如下：

class Solution {
public:
    int divide(int dividend, int divisor) {
        if (!divisor || (dividend == INT_MIN && divisor == -1))
            return INT_MAX;
            
        int flag = (dividend < 0) ^ (divisor < 0) ? -1 : 1;
        int res = 0;
        
        long long ldividend = labs(dividend);
        long long ldivisor  = labs(divisor);
        
        unsigned int time = 1;
        
        while((ldivisor << 1) <= ldividend) {
            ldivisor <<= 1;
            time <<= 1;
        }
        
        while(ldivisor >= 1) {
            if(ldividend >= ldivisor) {
                ldividend -= ldivisor;
                res += time;
            }
            ldivisor >>= 1;
            time >>= 1;
        }
        
        return res * flag;
    }
};

23 Merge k Sorted Lists

Merge k Sorted Lists Total Accepted: 47008 Total Submissions: 223015 My Submissions Question Solution Merge k sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

使用最小堆，先将每个链表的头节点构成堆，每次从堆顶取出节点，将该节点的下一个节点加入堆，同时将该节点插入新链表的尾部，循环直至堆为空。代码如下：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
private:
    static bool _comp(ListNode *n1, ListNode *n2) {
        if(n1->val < n2->val)
            return false;
        else
            return true;
    }
    
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        vector<ListNode*> myHeap;
        ListNode *newHead = nullptr;
        ListNode *newTail = nullptr;
        
        for(auto list : lists) {
            if(list != nullptr) {
                myHeap.push_back(list);
            }
        }
        
        make_heap(myHeap.begin(), myHeap.end(), _comp);
        
        while(!myHeap.empty()) {
            ListNode *newNode = myHeap.front();
            pop_heap(myHeap.begin(), myHeap.end(), _comp);
            myHeap.pop_back();
            
            if(newNode->next != nullptr) {
                myHeap.push_back(newNode->next);
                push_heap(myHeap.begin(), myHeap.end(), _comp);
            }
            
            newNode->next = nullptr;
            
            if(newHead == nullptr) {
                newHead = newNode;
                newTail = newNode;
            }
            else {
                newTail->next = newNode;
                newTail = newTail->next;
            }
        }
        
        return newHead;
    }
};
